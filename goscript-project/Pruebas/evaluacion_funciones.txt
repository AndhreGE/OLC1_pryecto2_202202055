func obtenerCodigoBase() string {
    return "P2-OLC1-FUNCIONES"
}

func sumarAjustada(a int, b int) int {
    var resultado int = a + b
    resultado = resultado + 5
    return resultado
}

func formarEtiqueta(nombre string, nivel int) string {
    return nombre + "-N" + nivel
}

func factorial(n int) int {
    if n == 0 {
        return 1
    }
    return n * factorial(n - 1)
}

func fibonacci(n int) int {
    if n == 0 {
        return 0
    } else if n == 1 {
        return 1
    }
    return fibonacci(n - 1) + fibonacci(n - 2)
}

func ackermann(m int, n int) int {
    if m == 0 {
        return n + 1
    } else if n == 0 {
        return ackermann(m - 1, 1)
    }
    return ackermann(m - 1, ackermann(m, n - 1))
}

func hanoi(n int, origen string, auxiliar string, destino string, movimientos []string) []string {
    if n == 1 {
        movimientos = append(movimientos, origen + "->" + destino)
        return movimientos
    }

    movimientos = hanoi(n - 1, origen, destino, auxiliar, movimientos)
    movimientos = append(movimientos, origen + "->" + destino)
    movimientos = hanoi(n - 1, auxiliar, origen, destino, movimientos)
    return movimientos
}

func sumarSlice(datos []int) int {
    var total int = 0
    for i, valor := range datos {
        total = total + valor
    }
    return total
}

func contarMayoresQue(datos []int, limite int) int {
    var cantidad int = 0
    for i, valor := range datos {
        if valor > limite {
            cantidad = cantidad + 1
        }
    }
    return cantidad
}

func concatenarTokens(tokens []string) string {
    var salida string = ""
    for i, valor := range tokens {
        if i == 0 {
            salida = salida + valor
        } else {
            salida = salida + "-" + valor
        }
    }
    return salida
}

func duplicarSlice(datos []int) []int {
    var respuesta []int
    for i, valor := range datos {
        respuesta = append(respuesta, valor * 2)
    }
    return respuesta
}

func crearSerie(inicio int, cantidad int) []int {
    var serie []int
    var actual int = inicio
    for i := 0; i < cantidad; i++ {
        serie = append(serie, actual)
        actual = actual + 1
    }
    return serie
}

func filtrarPares(datos []int) []int {
    var pares []int
    for i, valor := range datos {
        if valor % 2 == 0 {
            pares = append(pares, valor)
        }
    }
    return pares
}
